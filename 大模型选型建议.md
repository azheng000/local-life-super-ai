## 三、技术实现

### 3.1 模型配置

```typescript
// src/config/models.ts

export const MODEL_CONFIG = {
  // 主力模型
  primary: {
    provider: 'deepseek',
    model: 'deepseek-chat',  // DeepSeek-V3
    apiKey: process.env.DEEPSEEK_API_KEY,
    baseURL: 'https://api.deepseek.com/v1',
    maxTokens: 4096,
    temperature: 0.7,
  },
  
  // 深度推理模型
  reasoning: {
    provider: 'deepseek',
    model: 'deepseek-reasoner',  // DeepSeek-R1
    apiKey: process.env.DEEPSEEK_API_KEY,
    baseURL: 'https://api.deepseek.com/v1',
    maxTokens: 8192,
    temperature: 0.3,
  },
  
  // 降级备选
  fallback: {
    provider: 'qwen',
    model: 'qwen-plus',
    apiKey: process.env.QWEN_API_KEY,
    baseURL: 'https://dashscope.aliyuncs.com/api/v1',
  },
}

// 场景路由规则
export const MODEL_ROUTING: Record<string, string> = {
  'intent_classification': 'primary',    // 意图分类
  'merchant_matching': 'primary',        // 商家匹配
  'agent_chat': 'primary',              // 智能体对话
  'story_generation': 'primary',        // 故事生成
  'complex_recommendation': 'reasoning', // 复杂推荐
}
```

### 3.2 模型调用服务

```typescript
// src/services/llm.service.ts

import OpenAI from 'openai'
import { MODEL_CONFIG, MODEL_ROUTING } from '../config/models'

export class LLMService {
  private clients: Map<string, OpenAI> = new Map()
  
  constructor() {
    // 初始化各模型的客户端
    Object.entries(MODEL_CONFIG).forEach(([name, config]) => {
      this.clients.set(name, new OpenAI({
        apiKey: config.apiKey,
        baseURL: config.baseURL,
      }))
    })
  }
  
  // 根据场景选择模型
  async chat(scene: string, messages: OpenAI.ChatCompletionMessageParam[]) {
    const modelName = MODEL_ROUTING[scene] || 'primary'
    const config = MODEL_CONFIG[modelName]
    const client = this.clients.get(modelName)!
    
    try {
      const response = await client.chat.completions.create({
        model: config.model,
        messages,
        max_tokens: config.maxTokens,
        temperature: config.temperature,
      })
      
      return response.choices[0].message.content
    } catch (error) {
      // 降级到备选模型
      console.error(`${modelName} 调用失败，降级到 fallback`)
      return this.fallbackChat(messages)
    }
  }
  
  // 降级调用
  private async fallbackChat(messages: OpenAI.ChatCompletionMessageParam[]) {
    const config = MODEL_CONFIG.fallback
    const client = this.clients.get('fallback')!
    
    const response = await client.chat.completions.create({
      model: config.model,
      messages,
    })
    
    return response.choices[0].message.content
  }
}
```
